-- Este código verifica se há no disco C:\ uma pasta chamada Bancos e se não 
-- houver ela será criada
DECLARE @result int
EXEC @result = xp_cmdshell 'dir C:\Bancos\Faculdade'
IF (@result <> 0)
   Exec master.dbo.xp_cmdshell 'MD C:\Bancos\Faculdade'
go
------------------------------------------------------------------
CREATE DATABASE Faculdade
ON PRIMARY
(
  name= 'Faculdade_primary',
  filename='C:\Bancos\Faculdade\Faculdade_primary1.mdf',
  size= 10MB,
  maxsize=UNLIMITED,
  filegrowth= 10MB
),
FILEGROUP DADOS
(
  name= 'Faculdade_Dados',
  filename='C:\Bancos\Faculdade\Faculdade_Dados1.ndf',
  size= 10MB,
  maxsize=UNLIMITED,
  filegrowth= 10MB
),
FILEGROUP INDICES
(
  name= 'Faculdade_Index',
  filename='C:\Bancos\Faculdade\Faculdade_indices1.ndf',
  size= 10MB,
  maxsize=UNLIMITED,
  filegrowth= 10MB
)
LOG ON
(
  name= 'Faculdade_Log',
  filename='C:\Bancos\Faculdade\Faculdade_Log1.ldf',
  size= 1MB,
  maxsize=10MB,
  filegrowth= 1MB
),
(
  name= 'Faculdade_Log2',
  filename='C:\Bancos\Faculdade\Faculdade_Log2.ldf',
  size= 1MB,
  maxsize=10MB,
  filegrowth= 1MB
)COLLATE Latin1_General_CI_AI_KS
go

go

Exec SP_HelpFile
go

go
Exec SP_HelpSort
go
SELECT * FROM ::Fn_HelpCollations()
WHERE name = 'Latin1_General_CI_AI_KS'
go

go

go
SELECT syscolumns.Name,
                 syscolumns.Collation 
FROM sysobjects,syscolumns
WHERE sysobjects.id = syscolumns.id
and sysobjects.Name = 'Aluno'
go
SELECT syscolumns.Name,
                 syscolumns.Collation 
FROM sysobjects,syscolumns
WHERE sysobjects.id = syscolumns.id
and sysobjects.Name = 'Treinamento'
go

CREATE NONCLUSTERED INDEX I_Aluno_1
ON Aluno(Nome_Aluno)
ON INDICES
go
CREATE NONCLUSTERED INDEX I_Trein_1
ON Treinamento(Nome_Trein)
ON INDICES
